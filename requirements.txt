# --- 필수 (웹 대시보드 + 모바일 페이지) ---------------------------------
flask                 # src.app: HTTP 서버
flask-sock            # src.app: /ws/camera raw WebSocket (양방향: 프레임·음성·랜드마크 ↔ 명령)
pyopenssl             # src.app: ssl_context="adhoc" 자체 서명 인증서 (getUserMedia에 HTTPS 필수)
opencv-python         # perception/src.perception_loop: 프레임 디코드, 문서 검출, 미리보기 창
numpy                 # opencv와 함께 사용
matplotlib            # kinematics.simulate / src.render: 3D 렌더링 (웹 /api/render 포함)
python-dotenv         # .env에서 DEVICE_NAME / GEMINI_API_KEY 로드
requests              # src.api.light: 라즈베리파이 조명 서버 호출 (병합으로 추가)

# --- 하드웨어 제어 (실제 서보를 움직일 때만) -----------------------------
dynamixel-sdk         # hardware: DYNAMIXEL Protocol 1.0 통신

# --- AI 기능 (없으면 해당 기능만 비활성) ---------------------------------
google-genai          # src.api.gemini: 채팅/요약/STT/문서 글자 읽기
mediapipe             # perception.body_tracker: 사람 몸(어깨) 인식 — 얼굴 인식
                      # 폴백으로 서버에서 직접 돈다. 얼굴/손 인식 자체는
                      # 휴대폰 브라우저가 MediaPipe Tasks Vision(JS)으로 하므로
                      # 이 패키지가 필요한 건 몸 폴백뿐이다(perception/body_tracker.py,
                      # perception/face_tracker.py, perception/hand_tracker.py 참고).
